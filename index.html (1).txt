<script>
function extractData() {
  const text = document.getElementById("inputText").value;

  // Extract emails
  const emailRegex = /[a-zA-Z0-9._%+-]+@[a-zA-Z0-9.-]+\.[a-z]{2,}/g;
  let emails = text.match(emailRegex) || [];
  emails = [...new Set(emails)].filter(email => validateEmail(email));

  // Extract phones
  const phoneRegex = /\+?\d[\d\s-]{7,}\d/g;
  let phones = text.match(phoneRegex) || [];
  phones = [...new Set(phones)];

  document.getElementById("emailOutput").value = emails.join("\n");
  document.getElementById("phoneOutput").value = phones.join("\n");

  if (!emails.length && !phones.length) alert("No emails or phone numbers found!");
}

function validateEmail(email) {
  const re = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  return re.test(email);
}

function copyToClipboard(id) {
  const element = document.getElementById(id);
  element.select();
  element.setSelectionRange(0, 99999);
  navigator.clipboard.writeText(element.value);
  alert("Copied to clipboard âœ…");
}

function downloadText(id, filename) {
  const text = document.getElementById(id).value;
  const blob = new Blob([text], { type: 'text/plain' });
  const link = document.createElement('a');
  link.href = URL.createObjectURL(blob);
  link.download = filename;
  link.click();
  URL.revokeObjectURL(link.href);
}
